FUNKCJE STRZAŁKOWE JAKO CALLBACKI Anonimowe funkcje strzałkowe świetnie nadają się jako callbacki dla iterujących metod tablicy z powodu krótszej składni deklarowania, szczególnie jeśli niepotrzebne jest ciało funkcji. const numbers = [5, 10, 15, 20, 25]; // Zadeklarowanie funkcji numbers.forEach(function (number, index) { console.log(`Indeks ${index}, wartość ${number}`); }); // Anonimowa funkcja strzałkowa numbers.forEach((number, index) => { console.log(`Indeks ${index}, wartość ${number}`); }); Funkcję strzałkową callback można również deklarować oddzielnie i przekazywać odnośnik do niej. Warto to robić, jeśli jedna funkcja jest wykorzystywana w kilku miejscach programu lub jeśli jest on duży. const numbers = [5, 10, 15, 20, 25]; const logMessage = (number, index) => { console.log(`Indeks ${index}, wartość ${number}`); }; numbers.forEach(logMessage);
